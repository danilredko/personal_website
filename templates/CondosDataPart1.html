This project shows how we get use BeautifulSoup library to get the data from the websites like condos.ca.


Let's parse the html code by using the url of the website: 


def simple_get(url):

    try:
        response = requests.get(url)
    except requests.exceptions.RequestException as e:
        print(e)
        sys.exit(1)
    html_soup = BeautifulSoup(response.text, 'html.parser')

    return html_soup


This function cleans the text from special characters like '\t', '\n'. etc. 

def clean_text(text):

    return text.replace('\t', '').replace('\n','').replace('$', '').replace(' ', '').replace(',', '').replace('#', '')


As all the data about the condos are stored in the tables, we can read each column and store it into the corresponding variables: 


def get_table(url):

    soup = simple_get(url)
    table = soup.find('table')
    x = len(table.findAll('tr'))
    table_data = []
    for row in table.findAll('tr')[1:x]:

        col = row.findAll('td')
        street_no = clean_text(col[0].getText())
        street = clean_text(col[1].getText())
        unit = clean_text(col[2].getText())
        price = clean_text(col[3].getText())
        sqft = clean_text(col[4].getText())
        beds = clean_text(col[5].getText())
        baths = clean_text(col[6].getText())
        parking = clean_text(col[7].getText())
        locker = clean_text(col[8].getText())
        table_data.append([street_no, street, unit, price, sqft, beds, baths, parking, locker])

    return table_data


Now, we can get the data from every page and store it into the variable: 


def get_all_data(url):

    pages = []
    for i in range(1, 42):

        page = get_table(url)
        pages = pages + page
        url = url.replace('page='+str(i), 'page='+str(i+1))

    return pages


When we are done, we write all the data into the file which can be used later :
